by:You
desc:Start alone. Discover your purpose.

Layout
*main

Resources
*phase
hidden
*intro
hidden
*introTick
hidden
*exploration
*sticks
*stone
*knowledge
*people
*workers
hidden
*materials
*tools
*energy
*path
hidden
*soldiers
*warMaterials
*warProgress
hidden

*systems
on tick:
if (intro=1)
if (introTick=1)
log Darkness...
yield 1 introTick
else if (introTick=2)
log You wake up.
yield 1 introTick
else if (introTick=3)
log Alone.
yield 1 introTick
else if (introTick=4)
log No memory.
yield 1 introTick
else if (introTick=5)
log Just instinct.
yield 1 introTick
else if (introTick=6)
log Survive.
yield 1 introTick
else if (introTick=7)
log Explore.
yield 1 introTick
else if (introTick=8)
log Adapt.
yield 1 introTick
else if (introTick=9)
log Become something more.
yield 1 introTick
else if (introTick=10)
phase is 1
intro is 2
end
end

if (warProgress>20)
if (soldiers<50)
log You lost the war.
else if (soldiers>100 and warMaterials>100)
log The world ends in nuclear fire.
else
log The war continues endlessly.
end
warProgress is 0
end
