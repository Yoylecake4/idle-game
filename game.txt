Let's make a game!
	name:Infinite Expedition
	author:You
	desc:A long journey from nothing to civilization.

Settings
	background:#0f172a

Layout
	*main
		contains:leftPanel, rightPanel

	*leftPanel
		contains:res, buttons, log
		class:left

	*rightPanel
		contains:store
		class:right

	*res
		contains:Resources

	*buttons
		contains:Buttons
		class:buttons

	*store
		contains:Buildings, Upgrades

	*log
		contains:Log

Buttons

	// ===== STORY PHASE =====
	*start
		name:Wake Up
		req:stage=0
		on click:
			log You wake up alone in a strange land.
			stage is 1
		end

	*look
		name:Look Around
		req:stage=1
		on click:
			log You see trees, rocks, and open land.
			stage is 2
		end

	*sticks
		name:Gather Sticks
		req:stage=2
		on click:
			yield 1 sticks
			log You gather sticks.
			if (sticks >= 5)
				stage is 3
			end
		end

	*camp
		name:Build Camp
		req:stage=3
		on click:
			log You build a camp. A new life begins.
			unlockGame is 1
		end

	// ===== MAIN GAME =====
	*work
		name:Work
		req:unlockGame=1
		on click:yield 1 energy

	*forage
		name:Forage
		req:unlockGame=1
		on click:yield 1 food

	*gather
		name:Gather Materials
		req:unlockGame=1
		on click:yield 1 materials

	*craft
		name:Craft Tool
		req:unlockGame=1 and materials>=2
		on click:
			yield -2 materials
			yield 1 tools
		end

Resources
	*stage
		hidden

	*unlockGame
		hidden

	// INVENTORY
	*sticks
		name:Sticks
		hidden when 0

	*tools
		name:Tools
		hidden when 0

	// MAIN
	*energy
		name:Energy
		hidden when 0

	*food
		name:Food
		hidden when 0

	*materials
		name:Materials
		hidden when 0

Buildings
	*workers
		name:Workers
		req:unlockGame=1
		cost:10 energy
		on tick:yield 1 energy

	*farms
		name:Farms
		req:unlockGame=1
		cost:20 materials
		on tick:yield 1 food

Upgrades
	*eff
		name:Efficiency
		req:unlockGame=1
		cost:20 tools
		passive:multiply yield of energy by 2
