Let's make a game!
	name:Infinite Expedition
	author:You
	desc:A long journey from nothing to a civilization.

Settings
	background:#0f172a

Layout
	*main
		contains:leftPanel, rightPanel

	*leftPanel
		contains:res, buttons, log
		class:left

	*rightPanel
		contains:store
		class:right

	*res
		contains:Resources

	*buttons
		contains:Buttons
		class:buttons

	*store
		contains:Buildings, Upgrades

	*log
		contains:Log

Buttons

	// === STORY PHASE ===
	*startJourney
		name:Wake Up
		on click:
			log You wake up alone in a strange land.
			yield 1 stage
		end

	*lookAround
		name:Look Around
		req:1 stage
		on click:
			log You see trees, rocks, and open land.
			yield 1 stage
		end

	*gatherSticks
		name:Gather Sticks
		req:2 stage
		on click:
			yield 1 sticks
			log You gather some sticks.
		end

	*makeCamp
		name:Build Camp
		req:5 sticks
		on click:
			log You build a small camp. Your journey begins.
			yield 1 unlockGame
		end

	// === MAIN GAME (LOCKED UNTIL CAMP) ===
	*work
		name:Work
		req:1 unlockGame
		on click:yield 1 energy

	*forage
		name:Forage
		req:1 unlockGame
		on click:yield 1 food

	*gather
		name:Gather Materials
		req:1 unlockGame
		on click:yield 1 materials

	*craftTools
		name:Craft Tool
		req:2 materials
		on click:
			yield -2 materials
			yield 1 tools
		end

Resources
	*stage
		hidden

	*unlockGame
		hidden

	// INVENTORY
	*sticks
		name:Sticks
		hidden when 0

	*tools
		name:Tools
		hidden when 0

	// MAIN RESOURCES
	*energy
		name:Energy
		hidden when 0

	*food
		name:Food
		hidden when 0

	*materials
		name:Materials
		hidden when 0

Buildings
	*workers
		name:Workers
		req:1 unlockGame
		cost:10 energy
		on tick:yield 1 energy

	*farms
		name:Farms
		req:1 unlockGame
		cost:20 materials
		on tick:yield 1 food

Upgrades
	*efficiency
		name:Efficiency
		req:1 unlockGame
		cost:50 tools
		passive:multiply yield of energy by 2

Achievements
	*begin
		name:A New Beginning
		req:1 unlockGame
